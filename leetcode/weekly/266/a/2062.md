先考虑一个简单的情况：$\textit{word}$ 中的字母全是元音。

做法和 [1358. 包含所有三种字符的子字符串数目](https://leetcode.cn/problems/number-of-substrings-containing-all-three-characters/) 是一样的，请先完成这题。[我的题解](https://leetcode.cn/problems/number-of-substrings-containing-all-three-characters/solutions/2990226/mo-ban-yue-chang-yue-he-fa-xing-hua-dong-2g7a/)。

本题 $\textit{word}$ 中有辅音，并且要求**子串只包含元音**。

我们可以在遍历 $\textit{word}$ 的同时，维护子串的最小左端点 $\textit{start}$（初始值为 $0$）。当遍历到辅音 $\textit{word}[i]$ 时，更新 $\textit{start} = i + 1$，表示后续子串的左端点不能小于 $i+1$，否则子串包含辅音 $\textit{word}[i]$，不满足要求。于是，当子串右端点固定时，满足要求的子串左端点的范围是 $[\textit{start},\textit{left}-1]$，这有 $\textit{left}-\textit{start}$ 个，加到答案中。

```py [sol-Python3]
class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        cnt = defaultdict(int)
        ans = start = left = 0

        for i, ch in enumerate(word):
            if ch not in "aeiou":
                cnt.clear()  # 重置
                start = left = i + 1
                continue

            # 做法类似 1358. 包含所有三种字符的子字符串数目（哈希表写法）
            cnt[ch] += 1
            while len(cnt) == 5:  # 窗口包含所有元音
                out = word[left]
                cnt[out] -= 1
                if cnt[out] == 0:
                    del cnt[out]
                left += 1
            ans += left - start

        return ans
```

```java [sol-Java]
class Solution {
    public int countVowelSubstrings(String word) {
        char[] s = word.toCharArray();
        Map<Character, Integer> cnt = new HashMap<>();
        int ans = 0;
        int start = 0;
        int left = 0;

        for (int i = 0; i < s.length; i++) {
            char ch = s[i];
            if ("aeiou".indexOf(ch) < 0) {
                cnt.clear(); // 重置
                start = left = i + 1;
                continue;
            }

            // 做法类似 1358. 包含所有三种字符的子字符串数目（哈希表写法）
            cnt.merge(ch, 1, Integer::sum); // cnt[ch]++
            while (cnt.size() == 5) { // 窗口包含所有元音
                char out = s[left];
                int c = cnt.merge(out, -1, Integer::sum); // c = --cnt[out]
                if (c == 0) {
                    cnt.remove(out);
                }
                left++;
            }
            ans += left - start;
        }

        return ans;
    }
}
```

```cpp [sol-C++]
class Solution {
    static constexpr string VOWELS = "aeiou";

public:
    int countVowelSubstrings(string word) {
        unordered_map<char, int> cnt;
        int ans = 0, start = 0, left = 0;

        for (int i = 0; i < word.size(); i++) {
            char ch = word[i];
            if (!VOWELS.contains(ch)) {
                cnt.clear(); // 重置
                start = left = i + 1;
                continue;
            }

            // 做法类似 1358. 包含所有三种字符的子字符串数目（哈希表写法）
            cnt[ch]++;
            while (cnt.size() == 5) { // 窗口包含所有元音
                char out = word[left];
                cnt[out]--;
                if (cnt[out] == 0) {
                    cnt.erase(out);
                }
                left++;
            }
            ans += left - start;
        }

        return ans;
    }
};
```

func countVowelSubstrings(word string) (ans int) {
	cnt := map[byte]int{}
	start, left := 0, 0

	for i, ch := range word {
		if strings.IndexRune("aeiou", ch) < 0 {
			clear(cnt) // 重置
			start, left = i+1, i+1
			continue
		}

		// 做法类似 1358. 包含所有三种字符的子字符串数目（哈希表写法）
		cnt[byte(ch)]++
		for len(cnt) == 5 { // 窗口包含所有元音
			out := word[left]
			cnt[out]--
			if cnt[out] == 0 {
				delete(cnt, out)
			}
			left++
		}

	}

}

#### 复杂度分析

- 时间复杂度：$\mathcal{O}(n)$ 或 $\mathcal{O}(n|\Sigma|)$，其中 $n$ 是 $\textit{word}$ 的长度，$|\Sigma|=5$ 是元音种类数。
- 空间复杂度：$\mathcal{O}(|\Sigma|)$。
